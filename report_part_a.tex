\section{数据清洗流程}
% ============================================================

数据清洗是机器学习流程中至关重要的一步。原始数据往往存在格式不一致、列名不规范、缺失值等问题，必须经过系统性的清洗才能用于模型训练。

\subsection{列名规范化}

\textbf{问题发现：} 在读取CSV文件后，发现部分列名存在前导或尾随空格，例如：
\begin{itemize}
    \item \texttt{' Total ERC20 tnxs'}（前导空格）
    \item \texttt{'max value received '}（尾随空格）
\end{itemize}

这种不一致会导致后续代码中列名匹配失败，引发\texttt{KeyError}。

\textbf{解决方案：}
\begin{lstlisting}[language=Python, caption=列名规范化代码]
def load_data(path='data/ethereum_fraud.csv'):
    df = pd.read_csv(path)
    # 去除空列名或Unnamed列
    df = df.loc[:, ~df.columns.str.contains('^Unnamed')]
    # strip列名前后空格，统一格式
    df.columns = df.columns.str.strip()
    return df
\end{lstlisting}

\textbf{效果：} 所有列名统一为无空格的规范格式，例如\texttt{'Total ERC20 tnxs'}。

\subsection{非特征列去除}

在构建特征矩阵时，需要去除以下列：

\begin{enumerate}
    \item \textbf{标识列：} \texttt{Address}（字符串ID，无预测价值）
    \item \textbf{标签列：} \texttt{FLAG}（目标变量，不应作为特征）
    \item \textbf{索引列：} \texttt{Index}（如存在）
    \item \textbf{文本列：} \texttt{ERC20 most sent token type}、\texttt{ERC20\_most\_rec\_token\_type}
\end{enumerate}

\begin{lstlisting}[language=Python, caption=特征矩阵构建]
def get_feature_matrix(df):
    drop_cols = ['Address', 'FLAG']
    if 'Index' in df.columns:
        drop_cols.append('Index')
    # 去掉文本列（token类型）
    text_cols = [
        'ERC20 most sent token type', 
        'ERC20_most_rec_token_type'
    ]
    for col in text_cols:
        if col in df.columns:
            drop_cols.append(col)
    
    X = df.drop(columns=[c for c in drop_cols 
                         if c in df.columns], 
                errors='ignore')
    return X
\end{lstlisting}

\section{特征工程详细设计}
% ============================================================

特征工程是提升模型性能的关键环节。本项目基于领域知识（区块链交易行为模式）设计了10个衍生特征。

\subsection{设计原则}

\begin{enumerate}
    \item \textbf{领域驱动：} 基于区块链交易的行为模式设计
    \item \textbf{可解释性：} 每个特征都有明确的业务含义
    \item \textbf{稳定性：} 避免除以0，使用$\epsilon = 10^{-6}$平滑
\end{enumerate}

\subsection{比率特征}

\subsubsection{发送接收比率（sent\_received\_ratio）}

\textbf{数学公式：}
\begin{equation}
\text{sent\_received\_ratio} = \frac{\text{Sent tnx}}{\text{Received Tnx} + \epsilon}
\end{equation}

\textbf{业务含义：} 衡量地址是" predominantly发送方"还是" predominantly接收方"。
\begin{itemize}
    \item 高比率：地址主要向外发送资金（可能为资金转出、洗钱行为）
    \item 低比率：地址主要接收资金（可能为收款方、诈骗受害方）
\end{itemize}

\textbf{为什么有效：} 欺诈地址往往有特定的资金流向模式，例如频繁接收小额资金后集中转出。

\subsubsection{平均交易金额（avg\_transaction\_value）}

\textbf{数学公式：}
\begin{equation}
\text{avg\_transaction\_value} = \frac{\text{total Ether sent}}{\text{Sent tnx} + \epsilon}
\end{equation}

\textbf{业务含义：} 每次发送交易的平均ETH金额。
\begin{itemize}
    \item 高平均值：可能为大额资金转移、机构地址
    \item 低平均值：可能为频繁小额测试交易、欺诈试探行为
\end{itemize}

\subsubsection{合约创建比率（contract\_ratio）}

\textbf{数学公式：}
\begin{equation}
\text{contract\_ratio} = \frac{\text{Number of Created Contracts}}{\text{total transactions} + \epsilon}
\end{equation}

\textbf{业务含义：} 交易中有多少比例是合约创建。
\begin{itemize}
    \item 高比率：开发者地址、DeFi项目部署方
    \item 低比率：普通用户地址
\end{itemize}

\textbf{欺诈关联：} 某些欺诈行为会创建恶意合约（如蜜罐合约），该特征可帮助识别。

\subsubsection{每笔交易余额（balance\_per\_transaction）}

\textbf{数学公式：}
\begin{equation}
\text{balance\_per\_transaction} = \frac{\text{total ether balance}}{\text{total transactions} + \epsilon}
\end{equation}

\textbf{业务含义：} 平均每次交易后的ETH余额水平。

% ============================================================
\subsection{时间特征}

\subsubsection{总活跃时间（total\_time\_active）}

\textbf{数学公式：}
\begin{equation}
\text{total\_time\_active} = \text{Time Diff between first and last (Mins)}
\end{equation}

\textbf{业务含义：} 地址从创建首笔交易到末笔交易的时间跨度，反映地址的"生命周期"。

\subsubsection{平均发送间隔（avg\_sent\_interval）}

直接采用原始特征\texttt{Avg min between sent tnx}，不做转换。该特征反映发送交易的频率模式。

\subsubsection{平均接收间隔（avg\_received\_interval）}

直接采用原始特征\texttt{Avg min between received tnx}。

% ============================================================
\subsection{ETH流动特征}

\subsubsection{ETH净流入（ether\_flow）}

\textbf{数学公式：}
\begin{equation}
\text{ether\_flow} = \text{total ether received} - \text{total Ether sent}
\end{equation}

\textbf{业务含义：} 地址的ETH净流入量。
\begin{itemize}
    \item 正值：地址ETH增加（资金流入）
    \item 负值：地址ETH减少（资金流出）
\end{itemize}

\textbf{欺诈关联：} 欺诈地址可能呈现"快速流入-快速流出"的模式，ether\_flow接近0但交易频繁。

\subsubsection{最大接收比率（max\_received\_ratio）}

\textbf{数学公式：}
\begin{equation}
\text{max\_received\_ratio} = \frac{\text{max value received}}{\text{avg val received} + \epsilon}
\end{equation}

\textbf{业务含义：} 最大单笔接收金额相对于平均值的倍数，反映接收金额的离散程度。

\begin{itemize}
    \item 高比率：存在异常大额单笔交易（可能为攻击收益、洗钱）
    \item 低比率：交易金额分布均匀
\end{itemize}

% ============================================================
\subsection{ERC20活跃度}

\subsubsection{ERC20活跃比率（erc20\_activity）}

\textbf{数学公式：}
\begin{equation}
\text{erc20\_activity} = \frac{\text{Total ERC20 tnxs}}{\text{total transactions} + \epsilon}
\end{equation}

\textbf{业务含义：} ERC20代币交易占总交易的比例。
\begin{itemize}
    \item 高比率：活跃的DeFi用户、代币交易者
    \item 低比率：仅使用ETH的基础用户
\end{itemize}

\subsection{特征工程代码实现}

\begin{lstlisting}[language=Python, caption=特征工程完整实现]
def engineer_features(df):
    df = df.copy()
    eps = 1e-6
    
    # 基础比率特征
    df['sent_received_ratio'] = (
        df['Sent tnx'] / (df['Received Tnx'] + eps)
    )
    df['avg_transaction_value'] = (
        df['total Ether sent'] / (df['Sent tnx'] + eps)
    )
    df['contract_ratio'] = (
        df['Number of Created Contracts'] / 
        (df['total transactions (including tnx to create contract'] + eps)
    )
    df['balance_per_transaction'] = (
        df['total ether balance'] / 
        (df['total transactions (including tnx to create contract'] + eps)
    )
    
    # 时间相关特征
    df['total_time_active'] = df['Time Diff between first and last (Mins)']
    df['avg_sent_interval'] = df['Avg min between sent tnx']
    df['avg_received_interval'] = df['Avg min between received tnx']
    
    # ETH流动特征
    df['ether_flow'] = (
        df['total ether received'] - df['total Ether sent']
    )
    df['max_received_ratio'] = (
        df['max value received'] / (df['avg val received'] + eps)
    )
    
    # ERC20活跃度
    df['erc20_activity'] = (
        df['Total ERC20 tnxs'] / 
        (df['total transactions (including tnx to create contract'] + eps)
    )
    
    # 填充NaN
    df = df.fillna(0)
    return df
\end{lstlisting}

                    corr_matrix.iloc[x, y]) 
                   for x, y in zip(*high_corr) if x != y and x < y]

\subsection{特征缩放讨论}

\textbf{为什么不需要标准化：}

Random Forest是基于树的模型，分裂决策仅依赖于特征的排序关系，而非绝对数值。因此：
\begin{itemize}
    \item 不需要对特征进行标准化/归一化
    \item 不同量纲的特征（交易次数 vs ETH金额）可以直接使用
    \item 节省了预处理步骤，降低了复杂度
\end{itemize}

这与基于距离或梯度的模型（如SVM、神经网络）形成鲜明对比。

\subsection{Random Forest原理}

\subsubsection{算法概述}

随机森林是一种\textbf{集成学习}方法，通过构建多棵决策树并汇总其预测结果来提高准确性和鲁棒性。

\textbf{核心思想：}
\begin{enumerate}
    \item \textbf{Bootstrap采样：} 从原始训练集中有放回地抽取$N$个样本，构建$N$棵树的训练集
    \item \textbf{特征随机选择：} 每棵树分裂时，随机选择$m$个特征（$m \ll M$）进行最优分裂
    \item \textbf{投票机制：} 分类任务采用多数投票，回归任务采用平均值
\end{enumerate}

\subsubsection{数学表述}

设训练集为$D = \{(x_i, y_i)\}_{i=1}^{n}$，其中$x_i \in \mathbb{R}^{d}$，$y_i \in \{0, 1\}$。

随机森林由$T$棵决策树$\{h_t(x)\}_{t=1}^{T}$组成，最终预测为：
\begin{equation}
H(x) = \text{majority\_vote}\{h_t(x)\}_{t=1}^{T}
\end{equation}

其中每棵树$h_t$在Bootstrap样本集$D_t$上训练，且分裂时仅从随机选择的$m$个特征中寻找最优分裂点。

\subsubsection{关键超参数}

\texttt{class\_weight} & 'balanced' & 自动处理类别不平衡 \\

\textbf{class\_weight='balanced'的含义：}
自动调整类别权重，使得：
\begin{equation}
\text{weight\_class} = \frac{n\_\text{samples}}{n\_\text{classes} \times n\_\text{samples\_in\_class}}
\end{equation}

这确保欺诈样本（少数类）在模型训练中获得更高的权重，缓解类别不平衡问题。

from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import (
    accuracy_score, precision_score, 
    recall_score, f1_score
)

class FraudDetector:
    def __init__(self, n_estimators=100, random_state=42):
        self.model = RandomForestClassifier(
            n_estimators=n_estimators,
            random_state=random_state,
            class_weight='balanced'  # 处理类别不平衡
        )
        self.metrics = {}
        self.is_trained = False
    
    def train(self, X_train, y_train):
        self.model.fit(X_train, y_train)
        self.is_trained = True
    
    def predict(self, X):
        return self.model.predict(X)
    
    def predict_proba(self, X):
        return self.model.predict_proba(X)
    
    def evaluate(self, X_test, y_test):
        y_pred = self.predict(X_test)
        self.metrics = {
            'accuracy': accuracy_score(y_test, y_pred),
            'precision': precision_score(y_test, y_pred, zero_division=0),
            'recall': recall_score(y_test, y_pred, zero_division=0),
            'f1': f1_score(y_test, y_pred, zero_division=0)
        }
        return self.metrics
    
    def get_feature_importance(self):
        importance = pd.DataFrame({
            'feature': self.model.feature_names_in_,
            'importance': self.model.feature_importances_
        }).sort_values('importance', ascending=False)
        return importance
\end{lstlisting}

\subsection{评估指标}

对于二分类欺诈检测任务，采用以下指标：

\subsubsection{准确率（Accuracy）}
\begin{equation}
\text{Accuracy} = \frac{TP + TN}{TP + TN + FP + FN}
\end{equation}

\subsubsection{精确率（Precision）}
\begin{equation}
\text{Precision} = \frac{TP}{TP + FP}
\end{equation}
衡量被预测为欺诈的地址中，真正是欺诈的比例。

\subsubsection{召回率（Recall）}
\begin{equation}
\text{Recall} = \frac{TP}{TP + FN}
\end{equation}
衡量所有真实欺诈地址中，被模型成功检测出的比例。

\subsubsection{F1分数}
\begin{equation}
F1 = 2 \times \frac{Precision \times Recall}{Precision + Recall}
\end{equation}
精确率和召回率的调和平均，综合衡量模型性能。

\subsubsection{特异度（Specificity）}
\begin{equation}
\text{Specificity} = \frac{TN}{TN + FP}
\end{equation}
衡量所有真实正常地址中，被正确识别的比例。

\section{实验结果与讨论}
% ============================================================

\subsection{核心发现}

\begin{enumerate}
    \item \textbf{模型性能优异：} 96.14\%的准确率在二分类任务中属于优秀水平，证明特征工程和模型选择的有效性。
    
    \item \textbf{特征重要性揭示模式：} 交易数量、时间间隔、ETH余额等基础特征占据主导地位，说明欺诈地址在交易行为上与正常地址有显著差异。
    
    \item \textbf{衍生特征有效：} \texttt{sent\_received\_ratio}、\texttt{avg\_transaction\_value}等衍生特征进入Top 10，证明领域知识驱动的特征工程是有价值的。
    
    \item \textbf{类别不平衡处理有效：} class\_weight='balanced'确保了模型不会偏向于多数类（正常地址），在保持高准确率的同时实现了高召回率。
\end{enumerate}

\subsection{局限性}

\begin{enumerate}
    \item \textbf{数据集局限：} 数据集为静态快照，不包含时序动态变化，无法检测"先正常后欺诈"的地址演化
    \item \textbf{标签质量：} FLAG标签可能包含噪声，部分"正常"地址可能实际上是未标记的欺诈地址
    \item \textbf{模型解释性有限：} 虽然提供了特征重要性，但具体的决策规则（如"交易数$>$X且余额$<$Y"）不够明确
    \item \textbf{泛化能力未知：} 模型在新数据上的泛化能力需要进一步验证
\end{enumerate}

\subsection{未来改进方向}

\begin{enumerate}
    \item \textbf{集成异常检测：} 与Part B的Isolation Forest结合，识别训练集中未标记的异常
    \item \textbf{时序特征：} 加入交易时间序列模式，检测行为突变
    \item \textbf{模型调优：} 尝试XGBoost、LightGBM等模型，进行超参数网格搜索
    \item \textbf{阈值调优：} 根据业务需求（宁可错杀不可放过 vs 宁可放过不可错杀）调整分类阈值
\end{enumerate}
